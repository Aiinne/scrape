let handler = async (m, { conn, text }) => {
if (!text) throw 'Masukkan Jumlah Limit yang akan di tabung ...'
if (global.db.data.users[m.sender].limit <= text) throw 'Limit tidak cukup untuk ditabung !!'

handler.command = /^nabunglimit$/i
